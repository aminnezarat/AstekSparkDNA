135     val genExons = sc.textFile(bedFile)

137       .map(r=>(r.array(4).trim,r.array(5).trim.toInt,r.array(0).trim,r.array(1).trim.toInt,r.array(2).trim.toInt,r.array(3).trim) ).toArray
