18 /**
19 * Object for computing basic sample statistics.
20 */
21 object SparkSeqStats {
22   /**
23    * Method for computing mean value of a sample.
24    * @param xs Sample sequence
25    * @return Mean value of a sample.
26    */
27   def mean(xs: Seq[Int]): Double = xs match {
28     case Nil => 0.0
29     case ys => ys.reduceLeft(_ + _) / ys.size.toDouble
30   }
31
32   /**
33    * Method for computing standard deviation of a sample.
34    * @param xs Sample sequence
35    * @param avg Mean value of a sample.
36    * @return Standard deviation of a sample.
37    */
38   def stddev(xs: Seq[Int], avg: Double): Double = xs match {
39     case Nil => 0.0
40     case ys => math.sqrt((0.0 /: ys) {
41       (a,e) => a + math.pow(e - avg, 2.0)
42     } / xs.size)
43   }
44
45 }
