23 object SparkSeqFisherCombProb extends Serializable{
24   /**
25    * Method for combining p-values from various tests.
26    * @param pvalArray Array of p-values from other statistical tests.
27    * @return Fisher's combined test statistics.
28    */
29   def computeStatistics(pvalArray:Array[Double] ) : (Double,Double) ={
30     var fisherStat = 0.0
31     for(v <- pvalArray){
32       fisherStat += -2*log(v)
33     }
34     //return tuple : fisher stat,degrees of freedom
35     return((fisherStat,2*pvalArray.length) )
36   }

38   /**
39    * Method for computing a combined p-value using Fisher's method.
40    * @param fStat Fisher's combined test statistics.
41    * @param dof Degrees of freedom basing on a number of combined p-values.
42    * @return Combined p-value.
43    */
44   def getPValue(fStat:Double,dof:Double) : Double ={
45     
46     var pvalue = 1.0
47     val chiSquareDist = new ChiSquaredDistribution(dof)
48     pvalue = chiSquareDist.cumulativeProbability(fStat)
49     if (pvalue > 0.5)
50       pvalue=1-pvalue
51     return(pvalue)

53   }
